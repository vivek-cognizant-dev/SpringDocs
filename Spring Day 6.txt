@ModelAttribute is use to fetch entire Form data (imp)
	
//Previous
	@RequestMapping("/reg")
	public String regPage(@RequestParam("foodId") int fid, @RequestParam("foodName") String fName, @RequestParam("foodRate") int fRate, Model model) {
		model.addAttribute("foodId", fid);
		model.addAttribute("foodName", fName);
		model.addAttribute("foodRate", fRate);
		return "Registration";
	}

//After
	@RequestMapping("/reg")

		System.out.println(f.getFoodId());
