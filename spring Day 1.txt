Point.java

package com.cts;

public class Point {
	int x;
	int y;
	
	public Point() {
		super();
	}

	public Point(int x, int y) {
		super();
		this.x = x;
		this.y = y;
	}
	
	public int getX() {
		return x;
	}
	public void setX(int x) {
		this.x = x;
	}
	public int getY() {
		return y;
	}
	public void setY(int y) {
		this.y = y;
	}

	@Override
	public String toString() {
		return "Point [x=" + x + ", y=" + y + "]";
	}
	
}

Triangle.java

package com.cts;

import java.util.ArrayList;
import java.util.List;


public class Triangle {
	String name;
	Point point;
	List<String> al = new ArrayList<String>();
	
	public Triangle() {
		super();
	}

	public Triangle(String name, Point point, List<String> al) {
		super();
		this.name = name;
		this.point = point;
		this.al = al;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public Point getPoint() {
		return point;
	}

	public void setPoint(Point point) {
		this.point = point;
	}

	public List<String> getAl() {
		return al;
	}

	public void setAl(List<String> al) {
		this.al = al;
	}

	@Override
	public String toString() {
		return "Triangle [name=" + name + ", point=" + point + ", al=" + al + "]";
	}	
}
